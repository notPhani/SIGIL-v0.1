Developer Quick Start

Installation

// Install
npm install @sigil/auth

// Register user
import { register } from '@sigil/auth';

const { publicKey } = await register({
  username: 'alice',
  password: 'secret123'
});

// Sign transaction
const signature = await sign({
  username: 'alice',
  password: 'secret123',
  message: 'transfer $100'
});

// Verify (server-side)
const isValid = verify(
  signature, 
  publicKey
);
